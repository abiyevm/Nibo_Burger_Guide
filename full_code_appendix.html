A) Round Trip 
Task Description 
Create a program where the robot moves approximately 0.8–1 meter forward, performs a 180-degree 
turn, and then returns to its starting position. 
Code 
#include <NIBOburger.h> 
 
#define CRUISE_SPEED      380 
#define TURN_SPEED        420 
#define RAMP_STEP         40 
#define RAMP_INTERVAL_MS  25 
#define DRIVE_MS          2200 
#define TURN_MS           1200 
 
enum State { 
  DRIVE_OUT, 
  STOPPING_A, 
  TURNING, 
  STOPPING_B, 
  DRIVE_BACK, 
  STOPPING_C, 
  DONE 
}; 
 
State state = DRIVE_OUT; 
 
int currentSpeed = 0; 
unsigned long stateStart = 0; 
 
bool rampToward(int target) { 
  static unsigned long lastStep = 0; 
  unsigned long now = millis(); 
 
  if (now - lastStep < RAMP_INTERVAL_MS) 
    return false; 
 
  lastStep = now; 
 
  if (currentSpeed < target) 
    currentSpeed = min(currentSpeed + RAMP_STEP, target); 
 
  else if (currentSpeed > target) 
    currentSpeed = max(currentSpeed - RAMP_STEP, target); 
 

  return (currentSpeed == target); 
} 
 
void setState(State s) { 
  state = s; 
  stateStart = millis(); 
} 
 
void setup() { 
  NIBOburger.begin(); 
  delay(800); 
  stateStart = millis(); 
} 
 
void loop() { 
 
  unsigned long elapsed = millis() - stateStart; 
 
  switch (state) { 
 
    case DRIVE_OUT: 
      rampToward(CRUISE_SPEED); 
      Engine.setSpeed(currentSpeed, currentSpeed); 
 
      if (elapsed >= DRIVE_MS) 
        setState(STOPPING_A); 
      break; 
 
    case STOPPING_A: 
      rampToward(0); 
      Engine.setSpeed(currentSpeed, currentSpeed); 
 
      if (currentSpeed == 0) 
        setState(TURNING); 
      break; 
 
    case TURNING: 
      Engine.setSpeed(-TURN_SPEED, TURN_SPEED); 
 
      if (elapsed >= TURN_MS) 
        setState(STOPPING_B); 
      break; 
 
    case STOPPING_B: 
      Engine.setSpeed(0, 0); 
      currentSpeed = 0; 
 
      if (elapsed >= 300) 
        setState(DRIVE_BACK); 
      break; 
 
    case DRIVE_BACK: 
      rampToward(CRUISE_SPEED); 
      Engine.setSpeed(currentSpeed, currentSpeed); 
 
      if (elapsed >= DRIVE_MS) 
        setState(STOPPING_C); 
      break; 
 

    case STOPPING_C: 
      rampToward(0); 
      Engine.setSpeed(currentSpeed, currentSpeed); 
 
      if (currentSpeed == 0) 
        setState(DONE); 
      break; 
 
    case DONE: 
      Engine.setSpeed(0, 0); 
      break; 
  } 
} 
Explanation 
This program uses a state machine to control the robot step by step. The robot first drives forward, 
then stops smoothly, rotates approximately 180 degrees, drives back to the starting point, and finally 
stops completely. 
The program avoids using long delay() functions and instead uses millis() for timing. This 
allows the robot to continue processing actions without freezing the whole program. 
The rampToward() function gradually increases or decreases motor speed, creating smoother 
acceleration and braking. This prevents sudden movements and improves stability during driving 
and turning. 
The State variable keeps track of the robot’s current action. When one action finishes, the program 
automatically switches to the next state and resets the internal timer. 
B) Playing with Motors 
Task Description 
Create a motor-control program that demonstrates the use of global variables, local variables, and 
adjustable motor power using buttons. 
Code 
#include <NIBOburger.h> 
 
#define DEFAULT_POWER 400 
#define STEP          200 
#define MAX_POWER     1024 
#define MIN_POWER    -1024 
 
int power; 
 
void setup() { 
  NIBOburger.begin(); 
  power = DEFAULT_POWER; 
} 
 

void loop() { 
 
  NIBOburger.checkVoltage(); 
 
  char c = NIBOburger.getKeyChar(); 
 
  if (c != 0) { 
 
    if (c == 'a') { 
      power = power + STEP; 
 
      if (power > MAX_POWER) 
        power = MAX_POWER; 
    } 
 
    if (c == 'B') { 
      power = DEFAULT_POWER; 
    } 
 
    if (c == 'c') { 
      power = power - STEP; 
 
      if (power < MIN_POWER) 
        power = MIN_POWER; 
    } 
 
    Engine.setPWM(power, power); 
  } 
 
  delay(10); 
} 
Explanation 
This program demonstrates basic motor power control using buttons and variables. 
A global variable called power stores the current motor speed value. At startup, the motor power is 
initialized with the default value of 400. 
The robot continuously checks for button input using getKeyChar(): 
• 
Button 1 increases the motor power by 200.  
• 
Button 2 resets the power back to the default value.  
• 
Button 3 decreases the motor power by 200.  
To protect the motors, the program limits the allowed power range between -1024 and 1024. 
The selected power value is then applied to both motors using: 
Engine.setPWM(power, power); 
The checkVoltage() function is also used to monitor the battery voltage and help prevent unsafe 
low-voltage operation.

D) Using Buttons

Code 
#include <NIBOburger.h> 
 
int mode; 
unsigned long lastFlash; 
int flashState; 
 
void setup() { 
  NIBOburger.begin(); 
 
  mode = 0; 
  lastFlash = 0; 
  flashState = 0; 
}

  NIBOburger.checkVoltage(); 
 
  char c = NIBOburger.getKeyChar(); 
 
  if (c != 0) { 
 
    if (c == 'a') 
      mode = (mode == 1) ? 0 : 1; 
 
    if (c == 'B') 
      mode = (mode == 2) ? 0 : 2; 
 
    if (c == 'c') 
      mode = (mode == 3) ? 0 : 3; 
 
    if (mode == 0) { 
      NIBOburger.setLed(LED1, 0); 
      NIBOburger.setLed(LED2, 0); 
      NIBOburger.setLed(LED3, 0); 
      NIBOburger.setLed(LED4, 0);

  } 
 

  unsigned long now = millis(); 
 
  unsigned long interval = 
      (mode == 3) ? 80 : 300; 
 
  if (mode != 0 && 
      now - lastFlash >= interval) { 
 
    lastFlash = now; 
    flashState = !flashState;

      case 1: 
        NIBOburger.setLed(LED1, flashState); 
        NIBOburger.setLed(LED2, 0); 
        NIBOburger.setLed(LED3, flashState); 
        NIBOburger.setLed(LED4, 0); 
        break; 
 
      case 2: 
        NIBOburger.setLed(LED1, 0); 
        NIBOburger.setLed(LED2, flashState); 
        NIBOburger.setLed(LED3, 0); 
        NIBOburger.setLed(LED4, flashState);

        NIBOburger.setLed(LED1, flashState); 
        NIBOburger.setLed(LED2, flashState); 
        NIBOburger.setLed(LED3, flashState); 
        NIBOburger.setLed(LED4, flashState); 
        break; 
    } 
  }

} 
 
Explanation 
This program creates different flashing LED patterns controlled by buttons. 
The variable mode stores the currently active flashing pattern: 
• 
Mode 1 flashes LED1 and LED3.  
• 
Mode 2 flashes LED2 and LED4.  
• 
Mode 3 flashes all LEDs quickly to create a police siren effect.  
Pressing the same button again disables the active mode and turns all LEDs off. 
The flashing system is implemented using millis() instead of long delays. This keeps the robot 
responsive while the LEDs continue flashing. 
The variable flashState alternates between ON and OFF states, creating the blinking effect. 

E) Flashing LEDs Animation 
Task Description 
Create a smooth LED animation where the light travels back and forth across the LEDs. 
Animation sequence: 
LED1 → LED2 → LED3 → LED4 → LED3 → LED2 → repeat 
Additional effects include: 
• 
Smooth continuous movement  
• 
Gradually increasing speed  
• 
Short all-LED flash after each cycle  
Code 
#include <NIBOburger.h> 
 
#define MIN_INTERVAL  60 
#define MAX_INTERVAL  300 
#define SPEED_STEP    5 
 
int sequence[] = {0, 1, 2, 3, 2, 1}; 
 
int seqLength  = 6; 
int seqIndex   = 0; 
 
int interval   = MAX_INTERVAL; 
int cycleCount = 0; 
 
unsigned long lastStep; 
 
void allOff() { 
  NIBOburger.setLed(LED1, 0); 
  NIBOburger.setLed(LED2, 0); 
  NIBOburger.setLed(LED3, 0); 
  NIBOburger.setLed(LED4, 0); 
} 
 
void allOn() { 
  NIBOburger.setLed(LED1, 1); 
  NIBOburger.setLed(LED2, 1); 
  NIBOburger.setLed(LED3, 1); 
  NIBOburger.setLed(LED4, 1); 
} 
 
void showStep(int index) { 
 
  allOff(); 
 
  switch (index) { 

 
    case 0: 
      NIBOburger.setLed(LED1, 1); 
      break; 
 
    case 1: 
      NIBOburger.setLed(LED2, 1); 
      break; 
 
    case 2: 
      NIBOburger.setLed(LED3, 1); 
      break; 
 
    case 3: 
      NIBOburger.setLed(LED4, 1); 
      break; 
  } 
} 
 
void flashAll() { 
  allOn(); 
  delay(80); 
 
  allOff(); 
  delay(80); 
} 
 
void setup() { 
  NIBOburger.begin(); 
 
  lastStep = millis(); 
  seqIndex = 0; 
  cycleCount = 0; 
} 
 
void loop() { 
 
  NIBOburger.checkVoltage(); 
 
  unsigned long now = millis(); 
 
  if (now - lastStep >= (unsigned long)interval) { 
 
    lastStep = now; 
 
    showStep(sequence[seqIndex]); 
 
    seqIndex++; 
 
    if (seqIndex >= seqLength) { 
 
      seqIndex = 0; 
      cycleCount++; 
 
      flashAll(); 
 
      if (interval > MIN_INTERVAL) { 
        interval = interval - SPEED_STEP; 
      } 

    } 
  } 
} 
 
Explanation 
This program creates a dynamic LED animation effect using an array-based sequence system. 
The LED order is stored inside the array: 
{0, 1, 2, 3, 2, 1} 
The program moves through the sequence one step at a time using millis() timing. Only one LED 
is active at each step, creating a smooth travelling-light effect. 
After each full animation cycle: 
• 
All LEDs flash briefly together.  
• 
The animation speed increases slightly.  
• 
The delay interval decreases by 5 ms.  
The animation gradually accelerates from 300 ms down to a minimum of 60 ms per step, producing 
a smoother and more visually attractive effect. 
The code is divided into smaller helper functions such as: 
• 
allOn()  
• 
allOff()  
• 
showStep()  
• 
flashAll()  
This makes the program cleaner, easier to understand, and simpler to maintain. 
 
 
 
 
 

NIBO Burger Line Follower — Code 
Explanation 
Project Overview 
This program implements a PID-controlled line-following robot using the NIBO Burger platform 
and Arduino 1.6.7. The robot uses three surface sensors to detect the line, calculates its position, and 
corrects the motor speeds in real time to stay on track.  
Code 
#include <NIBOburger.h> 
#include <PID_v1.h> 
 
const int16_t BASE_SPEED   = 500; 
const int16_t MAX_PWM      = 900; 
const int16_t MIN_PWM      = 0; 
const double  KP           = 3.0; 
const double  KI           = 0.0; 
const double  KD           = 1.0; 
 
enum Position { 
  CENTER         = 1, 
  CENTER_RIGHT   = 2, 
  CENTER_LEFT    = 3, 
  HARD_RIGHT     = 4, 
  HARD_LEFT      = 5, 
  RECOVERY_RIGHT = 6, 
  RECOVERY_LEFT  = 7, 
  LOST           = 8, 
}; 
 
Position POSITION     = LOST; 
Position POSITION_OLD = LOST; 
 
float l_max = 1.0, c_max = 1.0, r_max = 1.0; 
 
double pid_input    = 0.0; 
double pid_output   = 0.0; 
double pid_setpoint = 0.0; 
 
PID steerPID(&pid_input, &pid_output, &pid_setpoint, KP, KI, KD, DIRECT); 
 
char key = ' '; 
 
float readL() { return l_max * SurfaceSensor.getL(0); } 
float readC() { return c_max * SurfaceSensor.getC(0); } 
float readR() { return r_max * SurfaceSensor.getR(0); } 
 
bool onLine(float val)  { return val < 45.0; } 
bool offLine(float val) { return val >= 45.0; } 
 
void calibrate_floor() { 

  POSITION = LOST; 
  NIBOburger.setLeds(1, 1, 1, 1); 
  key = ' '; 
 
  while (key != 'A') key = NIBOburger.getKeyChar(); 
  NIBOburger.setLeds(0, 0, 0, 0); 
 
  c_max = 100.0 / SurfaceSensor.getC(0); 
  l_max = 100.0 / SurfaceSensor.getL(0); 
  r_max = 100.0 / SurfaceSensor.getR(0); 
 
  while (key != 'B') key = NIBOburger.getKeyChar(); 
 
  NIBOburger.setLeds(1, 0, 1, 0); 
 
  while (key != 'C') { 
    key = NIBOburger.getKeyChar(); 
    NIBOburger.waitAnalogUpdate(); 
  } 
 
  key = ' '; 
} 
 
void get_position() { 
  float l = readL(); 
  float c = readC(); 
  float r = readR(); 
 
  pid_input = (double)(r - l); 
 
  bool L = onLine(l); 
  bool C = onLine(c); 
  bool R = onLine(r); 
 
  if (!L && C && !R) { 
    POSITION = CENTER; 
  }  
  else if (L && C && !R) { 
    POSITION = CENTER_LEFT; 
  }  
  else if (!L && C && R) { 
    POSITION = CENTER_RIGHT; 
  }  
  else if (L && !C && !R) { 
    POSITION = HARD_LEFT; 
  }  
  else if (!L && !C && R) { 
    POSITION = HARD_RIGHT; 
  }  
  else if (L && C && R) { 
    POSITION = CENTER; 
  }  
  else { 
    if (POSITION == HARD_RIGHT || POSITION == RECOVERY_RIGHT) { 
      POSITION = RECOVERY_RIGHT; 
    }  
    else { 
      POSITION = RECOVERY_LEFT; 
    } 

  } 
} 
 
void follow_line() { 
  int16_t pwm_left, pwm_right; 
 
  switch (POSITION) { 
 
    case RECOVERY_LEFT: 
      Engine.setPWM(-300, 300); 
      return; 
 
    case RECOVERY_RIGHT: 
      Engine.setPWM(300, -300); 
      return; 
 
    case HARD_LEFT: 
      Engine.setPWM(-200, BASE_SPEED); 
      return; 
 
    case HARD_RIGHT: 
      Engine.setPWM(BASE_SPEED, -200); 
      return; 
 
    default: 
      steerPID.Compute(); 
 
      pwm_left  = (int16_t)constrain(BASE_SPEED + pid_output, MIN_PWM, MAX_PWM); 
      pwm_right = (int16_t)constrain(BASE_SPEED - pid_output, MIN_PWM, MAX_PWM); 
 
      Engine.setPWM(pwm_left, pwm_right); 
      return; 
  } 
} 
 
void position_leds() { 
  if (POSITION_OLD == POSITION) return; 
 
  switch (POSITION) { 
 
    case CENTER: 
      NIBOburger.setLeds(0, 1, 1, 0); 
      break; 
 
    case CENTER_RIGHT: 
      NIBOburger.setLeds(0, 0, 1, 0); 
      break; 
 
    case CENTER_LEFT: 
      NIBOburger.setLeds(0, 1, 0, 0); 
      break; 
 
    case HARD_RIGHT: 
      NIBOburger.setLeds(0, 0, 1, 1); 
      break; 
 
    case HARD_LEFT: 
      NIBOburger.setLeds(1, 1, 0, 0); 
      break; 

 
    case RECOVERY_RIGHT: 
      NIBOburger.setLeds(0, 0, 0, 1); 
      break; 
 
    case RECOVERY_LEFT: 
      NIBOburger.setLeds(1, 0, 0, 0); 
      break; 
 
    case LOST: 
      NIBOburger.setLeds(1, 0, 0, 1); 
      break; 
  } 
 
  POSITION_OLD = POSITION; 
} 
 
void setup() { 
  NIBOburger.begin(); 
  SurfaceSensor.begin(); 
  Engine.begin(); 
 
  steerPID.SetMode(AUTOMATIC); 
  steerPID.SetSampleTime(20); 
  steerPID.SetOutputLimits(-400, 400); 
 
  calibrate_floor(); 
 
  delay(500); 
  Engine.setPWM(BASE_SPEED, BASE_SPEED); 
} 
 
void loop() { 
  NIBOburger.waitAnalogUpdate(); 
  NIBOburger.checkVoltage(); 
 
  get_position(); 
  follow_line(); 
  position_leds(); 
} 
Explanation 
1. Global Constants and PID Setup 
At the beginning of the program, the main control values are defined. These include the base speed, 
maximum PWM value, minimum PWM value, and PID tuning constants. 
The robot uses the following PID values: 
• 
KP = 3.0  
• 
KI = 0.0  
• 
KD = 1.0  

The PID controller calculates steering correction based on the difference between the right and left 
sensor readings. This difference is stored in pid_input. 
The PID output is limited between -400 and 400. This prevents the robot from overcorrecting too 
strongly when it moves at higher speed. 
2. Position States 
The program uses an enum called Position to describe where the robot is compared to the line. 
The possible states are: 
• 
CENTER  
• 
CENTER_RIGHT  
• 
CENTER_LEFT  
• 
HARD_RIGHT  
• 
HARD_LEFT  
• 
RECOVERY_RIGHT  
• 
RECOVERY_LEFT  
• 
LOST  
This makes the program easier to understand because each movement situation has a clear name. 
For example, if the robot is exactly on the line, the state is CENTER. If only the right sensor detects 
the line, the state becomes HARD_RIGHT, meaning the robot needs a stronger correction. 
3. Sensor Reading and Calibrattion 
The robot uses three surface sensors: 
• 
Left sensor  
• 
Center sensor  
• 
Right sensor  
Before the robot starts following the line, it performs calibration using the calibrate_floor() 
function. 
During calibration, the robot measures the floor surface and calculates correction values for each 
sensor: 
c_max = 100.0 / SurfaceSensor.getC(0); 
l_max = 100.0 / SurfaceSensor.getL(0); 
r_max = 100.0 / SurfaceSensor.getR(0); 
This makes the sensor readings more reliable under different lighting conditions and surface colors. 
The functions readL(), readC(), and readR() return normalized values from the left, center, and 
right sensors. 

4. Line Detection 
The program decides whether a sensor is on the line using this condition: 
bool onLine(float val) { 
  return val < 45.0; 
} 
If the sensor value is lower than 45, the program considers that sensor to be detecting the line. 
The opposite function is: 
bool offLine(float val) { 
  return val >= 45.0; 
} 
This threshold helps the robot understand whether it is on the line, slightly off-center, or completely 
lost. 
5. Position Detection 
The get_position() function reads all three sensors and decides the robot’s current position. 
For example: 
if (!L && C && !R) { 
  POSITION = CENTER; 
} 
This means only the center sensor sees the line, so the robot is correctly aligned. 
Other cases are used for left and right corrections: 
• 
Left and center sensors detect the line → CENTER_LEFT  
• 
Center and right sensors detect the line → CENTER_RIGHT  
• 
Only left sensor detects the line → HARD_LEFT  
• 
Only right sensor detects the line → HARD_RIGHT  
If no sensor detects the line, the robot enters recovery mode. It remembers the last strong direction 
and turns that way to search for the line again. 
This is important because it prevents the robot from randomly turning in the wrong direction after 
losing the line. 
6. Motor Control and PID Steering 
The follow_line() function controls the motors depending on the current position state. 

For hard turns and recovery situations, the robot uses fixed motor values. 
Example: 
case HARD_LEFT: 
  Engine.setPWM(-200, BASE_SPEED); 
  return; 
This makes the robot turn sharply to the left. 
For normal line-following situations, the PID controller calculates a correction: 
steerPID.Compute(); 
Then the correction is applied to the motors: 
pwm_left  = BASE_SPEED + pid_output; 
pwm_right = BASE_SPEED - pid_output; 
If the robot moves away from the line, one motor becomes faster and the other becomes slower. This 
steers the robot smoothly back toward the line. 
The constrain() function keeps motor values inside the safe range between MIN_PWM and MAX_PWM. 
7. LED Feedback 
The position_leds() function shows the robot’s current state using the onboard LEDs. 
Each position state has a different LED pattern. 
Examples: 
• 
CENTER lights the two middle LEDs.  
• 
HARD_LEFT lights the two left LEDs.  
• 
HARD_RIGHT lights the two right LEDs.  
• 
LOST lights the outer LEDs.  
This is useful during testing because the robot can show what it is detecting without using the Serial 
Monitor. 
The function also checks whether the position changed before updating LEDs: 
if (POSITION_OLD == POSITION) return; 
This avoids unnecessary LED updates and prevents flickering. 
 

8. Setup Function 
The setup() function prepares the robot before the main program starts. 
It initializes: 
• 
The NIBO Burger system  
• 
Surface sensors  
• 
Motors  
• 
PID controller  
The PID controller is set to automatic mode: 
steerPID.SetMode(AUTOMATIC); 
The sample time is set to 20 milliseconds: 
steerPID.SetSampleTime(20); 
After that, the robot performs floor calibration and then starts moving with the base speed. 
9. Main Loop 
The loop() function is kept short and clean. 
Each cycle performs four main actions: 
NIBOburger.waitAnalogUpdate(); 
NIBOburger.checkVoltage(); 
 
get_position(); 
follow_line(); 
position_leds(); 
First, the robot waits for updated sensor values. Then it checks the battery voltage for safety. 
After that, it detects the current position, controls the motors, and updates the LEDs. 
Keeping the loop simple makes the program easier to debug and modify. 
 
 
Conclusion 

This line follower program combines sensor calibration, PID control, position detection, motor 
correction, and LED feedback. 
The robot can follow a line smoothly using PID steering, but it can also make stronger corrections 
when it reaches sharp turns or loses the line. Recovery states help the robot search for the line in the 
correct direction instead of turning randomly. 
Overall, the program is structured clearly, with separate functions for calibration, sensor reading,

NIBO Burger Maze Solver — Code 
Explanation 
Project Overview 
This program implements a PID-controlled maze-solving robot using the NIBO Burger platform. 
The robot follows a line, detects junctions, chooses the correct direction using a left-hand priority 
rule, and continues until it reaches the finish. In testing, the robot solved the maze in 46 seconds.  
Code 
#include <NIBOburger.h> 
#include <PID_v1.h> 
 
const int16_t BASE_SPEED = 680; 
const int16_t MAX_PWM = 900; 
const int16_t MIN_PWM = 100;

const double KI = 0.0; 
const double KD = 1.2; 
 
const uint16_t JUNCTION_CONFIRM_MS = 60;

const uint16_t JUNCTION_SCAN_MS = 180; 
 
const uint16_t TURN_90_MS = 320; 
const int16_t TURN_SPEED = 420; 
 
const float JUNCTION_THRESHOLD = 35.0; 
const float STRAIGHT_THRESHOLD = 55.0; 
 
const uint8_t FINISH_JUNCTION = 99; 
 
struct Paths { 
  bool left; 
  bool straight; 

  bool right; 
};

  CENTER = 1, 
  CENTER_RIGHT, 
  CENTER_LEFT, 
  HARD_RIGHT, 
  HARD_LEFT, 
  RECOVERY_RIGHT, 
  RECOVERY_LEFT, 
  LOST 
}; 
 
enum MazeState { 
  MAZE_FOLLOWING, 
  MAZE_JUNCTION, 
  MAZE_FINISHED 
}; 
 
Position POSITION = LOST; 
Position POSITION_OLD = LOST; 
 
MazeState mazeState = MAZE_FOLLOWING; 
 
uint8_t junctionCount = 0; 
uint32_t lastJunctionMs = 0;

float c_max = 1.0; 
float r_max = 1.0;

PID steerPID( 
  &pid_input, 
  &pid_output, 
  &pid_setpoint, 
  KP, 
  KI, 
  KD, 
  DIRECT

  POSITION = LOST;

void get_position() { 
  float l = readL(); 
  float c = readC(); 
  float r = readR(); 
 
  pid_input = (double)(r - l); 
 
  bool L = onLine(l); 
  bool C = onLine(c); 
  bool R = onLine(r); 
 
  if (!L && C && !R) { 
    POSITION = CENTER; 
  } 
  else if (L && C && !R) { 
    POSITION = CENTER_LEFT; 
  } 

  else if (!L && C && R) { 
    POSITION = CENTER_RIGHT; 
  } 
  else if (L && !C && !R) { 
    POSITION = HARD_LEFT; 
  } 
  else if (!L && !C && R) { 
    POSITION = HARD_RIGHT; 
  } 
  else if (L && C && R) { 
    POSITION = CENTER; 
  } 
  else { 
    POSITION = 
      (POSITION == HARD_RIGHT || POSITION == RECOVERY_RIGHT) 
      ? RECOVERY_RIGHT 
      : RECOVERY_LEFT; 
  } 
} 
 
void follow_line_step() { 
  switch (POSITION) { 
 
    case RECOVERY_LEFT: 
      Engine.setPWM(-350, 350); 
      break; 
 
    case RECOVERY_RIGHT: 
      Engine.setPWM(350, -350); 
      break; 
 
    case HARD_LEFT: 
      Engine.setPWM(-150, BASE_SPEED); 
      break; 
 
    case HARD_RIGHT: 
      Engine.setPWM(BASE_SPEED, -150); 
      break; 
 
    default: 
      steerPID.Compute(); 
 
      Engine.setPWM( 
        constrain((int16_t)(BASE_SPEED + pid_output), MIN_PWM, MAX_PWM), 
        constrain((int16_t)(BASE_SPEED - pid_output), MIN_PWM, MAX_PWM) 
      ); 
      break; 
  } 
} 
 
bool junctionDetected() { 
  if (millis() - lastJunctionMs < POST_JUNCTION_GUARD_MS) { 
    return false; 
  } 
 
  static uint32_t firstSeenMs = 0; 
 
  float l = readL(); 
  float c = readC(); 

  float r = readR(); 
 
  bool leftArm = isJunctionArm(l); 
  bool rightArm = isJunctionArm(r); 
  bool centerOn = onLine(c); 
 
  bool isJunction = centerOn && (leftArm || rightArm); 
 
  if (isJunction) { 
 
    if (firstSeenMs == 0) { 
      firstSeenMs = millis(); 
    } 
 
    if (millis() - firstSeenMs >= JUNCTION_CONFIRM_MS) { 
      firstSeenMs = 0; 
      return true; 
    } 
  } 
  else { 
    firstSeenMs = 0; 
  } 
 
  return false; 
} 
 
Paths scanJunction() { 
  Paths p = {false, false, false}; 
 
  uint32_t start = millis(); 
 
  uint8_t leftCount = 0; 
  uint8_t rightCount = 0; 
  uint8_t centerCount = 0; 
 
  while (millis() - start < JUNCTION_SCAN_MS) { 
 
    Engine.setPWM(BASE_SPEED, BASE_SPEED); 
    NIBOburger.waitAnalogUpdate(); 
 
    float l = readL(); 
    float c = readC(); 
    float r = readR(); 
 
    if (isJunctionArm(l)) { 
      leftCount++; 
    } 
 
    if (isJunctionArm(r)) { 
      rightCount++; 
    } 
 
    if (onLine(c)) { 
      centerCount++; 
    } 
  } 
 
  uint8_t minHits = 2; 
 

  p.left = (leftCount >= minHits); 
  p.right = (rightCount >= minHits); 
  p.straight = (centerCount >= (JUNCTION_SCAN_MS / 10)); 
 
  Engine.setPWM(BASE_SPEED, BASE_SPEED); 
  delay(20); 
 
  Engine.setPWM(0, 0); 
 
  return p; 
} 
 
void doTurnLeft() { 
  Engine.setPWM(-TURN_SPEED, TURN_SPEED); 
  delay(TURN_90_MS); 
  Engine.setPWM(0, 0); 
} 
 
void doTurnRight() { 
  Engine.setPWM(TURN_SPEED, -TURN_SPEED); 
  delay(TURN_90_MS);

void doUTurn() { 
  Engine.setPWM(TURN_SPEED, -TURN_SPEED);

} 
 
void finishRoutine() { 
  Engine.setPWM(0, 0); 
 
  while (true) { 
    NIBOburger.setLeds(1, 1, 1, 1); 
    delay(300); 
 
    NIBOburger.setLeds(0, 0, 0, 0); 
    delay(300); 
  } 
}

void loop() { 
 
  NIBOburger.waitAnalogUpdate(); 
  NIBOburger.checkVoltage(); 
 
  switch (mazeState) { 
 
    case MAZE_FOLLOWING: 
 
      get_position(); 
      follow_line_step(); 
      position_leds(); 
 
      if (junctionDetected()) { 
        mazeState = MAZE_JUNCTION; 
      } 

 
      break; 
 
    case MAZE_JUNCTION: { 
 
      junctionCount++; 
      lastJunctionMs = millis(); 
 
      if (junctionCount >= FINISH_JUNCTION) { 
        mazeState = MAZE_FINISHED; 
        break; 
      } 
 
      Paths p = scanJunction(); 
 
      if (p.left) { 
        doTurnLeft(); 
        NIBOburger.setLeds(1, 0, 0, 0); 
      } 
      else if (p.straight) { 
        NIBOburger.setLeds(0, 1, 1, 0); 
      } 
      else if (p.right) { 
        doTurnRight(); 
        NIBOburger.setLeds(0, 0, 0, 1); 
      } 
      else { 
        doUTurn(); 
        NIBOburger.setLeds(1, 0, 0, 1); 
      } 
 
      mazeState = MAZE_FOLLOWING; 
 
      break; 
    } 
 
    case MAZE_FINISHED: 
      finishRoutine(); 
      break; 
  }
